```python
def solution(table, languages, preference):
    answer = ''
    rscore = 0
    for item in table:
        jd, *jd_langs = item.split()
        score = 0
        for k, lang in enumerate(languages):
            for i in range(len(jd_langs)):
                if jd_langs[i] == lang:
                    score += (5-i) * preference[k]

        if rscore < score:
            rscore = score
            answer = jd

        elif rscore == score and answer > jd: answer = jd

    return answer
```
