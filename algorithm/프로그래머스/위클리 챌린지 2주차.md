```python
def get_credit(avg):
    if avg >= 90: return 'A'
    elif avg >= 80: return 'B'
    elif avg >= 70: return 'C'
    elif avg >= 50: return 'D'
    else: return 'F'

def solution(scores):
    answer = ''
    n = len(scores)
    
    for i in range(n):
        tmp = [scores[x][i] for x in range(n) if i != x]
        avg = 0
        if min(tmp) <= scores[i][i] <= max(tmp): avg = (sum(tmp) + scores[i][i]) / n
        else: avg = sum(tmp) / (n-1)
        answer += get_credit(avg)
        
    return answer
```
